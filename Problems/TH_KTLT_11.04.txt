#include <stdio.h>

// tim ucln
// dau ra ucln, -1 loi 
int gcd(int a, int b) {
	if ((a > 0) && (b > 0)) {
		// neu a  <= b thi doi gia tri
		if (a <= b) { // doi gia tri a va b
			int tmp = a;
			a = b;
			b = tmp;
		}
		if (b == 0) {
			printf("KQ: %d\n", a);
			return a;
		}
		int r;
		while (b != 0) {
			r = a % b;
			a = b;
			b = r;
		}
		printf("KQ: %d\n", a);
		return a;

	}
	else {
		printf("Dau vao k hop le \n");
		return -1;
	}
}

int main() {
	int a, b;

	printf("Nhap 2 so a, b: ");
	scanf_s("%d", &a);
	scanf_s("%d", &b);

	int kq = gcd(a, b);
}

#include <stdio.h>
#include <math.h>

int main() {
    int n;
    int count = 0;
    printf("\nNhap n = ");
    scanf_s("%d", &n);
    if (n == 1 ) {
        printf("\n%d la so nguyen to", n);
        return 0;
    }
    for (int i = 2; i <= sqrt(n); i++) {
        if (n % i == 0) {
            count++;
        }
    }
    if (count == 0) {
        printf("\n%d la so nguyen to", n);
    }
    else {
        printf("\n%d khong phai so nguyen to", n);
    }
    return 0;
}
