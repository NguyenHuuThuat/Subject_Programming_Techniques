void nhapSo(int x[]) {

		i = i + 1;
	}

// dau vao 2 mang ban dau
// dau ra 2 mang dao nhau
int soDao(int x) {
	int y = 0;
	
	while (x > 0)
	{
		y = 10 * y + (x % 10);
		x = x / 10;
	}
	return y;
}

// tim ucln
// dau ra ucln, -1 loi 
int ucln(int a, int b) {
	if ((a > 0) && (b > 0)) {
		// neu a  <= b thi doi gia tri
		if (a <= b) { // doi gia tri a va b
			int tmp = a;
			a = b;
			b = tmp;
		}
		if (b == 0) {
			//printf("KQ: %d\n", a);
			return a;
		}
		int r;
		while (b != 0) {
			r = a % b;
			a = b;
			b = r;
		}
		//printf("KQ: %d\n", a);
		return a;
	}
	else {
		//printf("Dau vao k hop le \n");
		return -1;
	}
}
bool NTCN(int x, int y) {
	int i;
	int kq;
	kq = ucln(x, y);
	if (kq == 1) {
		return true;
	}
	else {
		return false;
	}

}

bool soDaoNTCN(int x) {
	if (NTCN (x, soDao(x))) {
		return true;
	}
	else
		return false;
}

int Tim(int s[]) {
	int x;
	int n = 0;
	for (x = 10; x < 100; x++) {
		if (soDaoNTCN(x)) {
			s[n] = x;
			n = n + 1;
		}
	}
	return n;
}

#include<stdio.h>


void NhapvaInPhanSo(float *TS, float *MS) {
	printf("Nhap tu so cua phan so: ");
	scanf_s("%f", TS);
	do
	{
		printf("Nhap mau so cua phan so: ");
		scanf_s("%f", MS);
	} while (*MS == 0);
	printf("Phan so la: %f/%f \n", *TS, *MS);
}

void Tong() {
	printf("Phep cong phan so: \n");
	float TS1, MS1, TS2, MS2, TSSum,MSSum;
	printf("Nhap phan so thu 1 : \n");
	NhapvaInPhanSo(&TS1, &MS1);
	printf("Nhap phan so thu 2 : \n");
	NhapvaInPhanSo(&TS2, &MS2);
	TSSum = TS1 * MS2 + TS2 * MS1;
	MSSum = MS1 * MS2;
	printf("KQ %f/%f\n", TSSum, MSSum);
}
